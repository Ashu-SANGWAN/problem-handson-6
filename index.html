Q. 1 Find the Smaller Angle

 
var Minimal_Angle = (h, m) => 
{
  let hourAngle = (30*h) + (0.5*m);
  let minsAngle = 6*m;
  let minimalAngle=0;
  if(h===12 && m===0)
  {
    return 0;
  }
  if(hourAngle>minsAngle)
  {
    minimalAngle = (hourAngle - minsAngle);
  }
  else
  {
     minimalAngle = (minsAngle - hourAngle);
  }
  return minimalAngle;
};

Q.3 Perfect number Check

 var Perfect_Check = (N) => 
    {
      let perfectSum=0;
      for(let i=0;i < N;i++)
      {
        if(N%i===0)
        {
          perfectSum+=i;
        }
      }
      if(perfectSum===N)
      {
        return "YES";
      }
      else
      {
        return "NO";
      }
    };
